WITH search_vector AS (
  SELECT VECTOR_EMBEDDING(
    ALL_MINILM_L12_V2 USING 'symptoms of sudden weakness in the left arm and leg' AS data
  ) AS vec FROM DUAL
)
SELECT
    m.doc_json,
    L2_DISTANCE(
        m.doc_vector,
        v.vec
    ) AS distance
FROM
    medical_qa_data m,
    search_vector v
ORDER BY
    distance
FETCH FIRST 5 ROWS ONLY;
